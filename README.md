# node-cli-contacts

# Get contact list (console.table)

node index.js --action list

link

# Get contact by id

node index.js --action get --id 5

link

# Add contact

node index.js --action add --name Mango --email mango@gmail.com --phone 322-22-22

link

# Remove contact

node index.js --action remove --id=3

link
